class Heap {
    constructor() {
        this.heap = new Array();
        this.heap.push(0);
    }

    push(val) {
        this.heap.push(val);
        let i = this.heap.length - 1;

        // Percolate up
        while (i > 1 && this.heap[i] < this.heap[Math.floor(i / 2)]) {
            let tmp = this.heap[i];
            this.heap[i] = this.heap[Math.floor(i / 2)];
            this.heap[Math.floor(i / 2)] = tmp;
            i = Math.floor(i / 2);
        }
    }

    pop() {

            return -1;
